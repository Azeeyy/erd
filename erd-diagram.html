üíª SQL Schema (Untuk phpMyAdmin)

-- Buat database
CREATE DATABASE db_pengaduan_sekolah;
USE db_pengaduan_sekolah;

-- Tabel users
CREATE TABLE users (
  user_id INT AUTO_INCREMENT PRIMARY KEY,
  username VARCHAR(50) NOT NULL,
  password VARCHAR(255) NOT NULL,
  nama VARCHAR(100) NOT NULL,
  email VARCHAR(100),
  role ENUM('admin','siswa','guru','staff') DEFAULT 'siswa',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Tabel kategori
CREATE TABLE kategori (
  kategori_id INT AUTO_INCREMENT PRIMARY KEY,
  nama_kategori VARCHAR(50) NOT NULL,
  deskripsi TEXT
);

-- Tabel pengaduan
CREATE TABLE pengaduan (
  pengaduan_id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT NOT NULL,
  kategori_id INT NOT NULL,
  judul VARCHAR(200) NOT NULL,
  deskripsi TEXT NOT NULL,
  status ENUM('pending','diproses','selesai','ditolak') DEFAULT 'pending',
  prioritas ENUM('rendah','sedang','tinggi') DEFAULT 'sedang',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(user_id),
  FOREIGN KEY (kategori_id) REFERENCES kategori(kategori_id)
);

-- Tabel bukti
CREATE TABLE bukti (
  bukti_id INT AUTO_INCREMENT PRIMARY KEY,
  pengaduan_id INT NOT NULL,
  file_path VARCHAR(255) NOT NULL,
  file_name VARCHAR(100) NOT NULL,
  file_size INT,
  uploaded_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (pengaduan_id) REFERENCES pengaduan(pengaduan_id)
);

-- Tabel tanggapan
CREATE TABLE tanggapan (
  tanggapan_id INT AUTO_INCREMENT PRIMARY KEY,
  pengaduan_id INT NOT NULL,
  admin_id INT NOT NULL,
  isi_tanggapan TEXT NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (pengaduan_id) REFERENCES pengaduan(pengaduan_id),
  FOREIGN KEY (admin_id) REFERENCES users(user_id)
);
